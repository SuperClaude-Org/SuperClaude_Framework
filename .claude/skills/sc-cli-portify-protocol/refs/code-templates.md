## File Generation Order

Generate in this order to respect dependencies:

1. `models.py` — No internal deps
2. `gates.py` — Imports from models
3. `prompts.py` — Imports from models
4. `config.py` — Imports from models
5. `monitor.py` — Imports from models
6. `process.py` — Imports from models, config; extends pipeline.process
7. `executor.py` — Imports from everything above
8. `tui.py` — Imports from models
9. `logging_.py` — Imports from models
10. `diagnostics.py` — Imports from models
11. `commands.py` — Imports from config, executor
12. `__init__.py` — Re-exports

    """Aggregate result for the entire {PIPELINE_NAME} pipeline."""

        end = self.finished_at or time.time()
        return end - self.started_at

        return sum(1 for r in self.step_results
                   if r.status in ({PREFIX}Status.PASS, {PREFIX}Status.PASS_NO_SIGNAL))

    def steps_failed(self) -> int:
        return sum(1 for r in self.step_results
                   if r.status in ({PREFIX}Status.HALT, {PREFIX}Status.ERROR, {PREFIX}Status.TIMEOUT))

        if self.halt_step:

    def stall_status(self) -> str:
        if self.events_received == 0:
            elapsed = time.time() - self.step_started_at if self.step_started_at else 0
            return "waiting..." if elapsed < 30 else "STALLED"
        if self.stall_seconds > 120:
            return "STALLED"
        if self.stall_seconds > 30:
            return "thinking..."
        return "active"

    @property
    def output_size_display(self) -> str:
        if self.output_bytes < 1024:
            return f"{self.output_bytes}B"
        return f"{self.output_bytes / 1024:.1f}KB"

## Template: `executor.py`

```python
"""Sprint-style supervised executor for {PIPELINE_NAME} pipeline.

Controls the execution loop with:
- Per-step live monitoring
- TUI refresh at ~2Hz
- Stall detection and watchdog
- Signal-aware graceful shutdown
- Deterministic status classification
- Failure diagnostics

IMPORTANT: Use synchronous execution with threading, NOT async/await.
Sprint uses time.sleep() polling loops and concurrent.futures.ThreadPoolExecutor
for parallel batch dispatch. Do not use asyncio.

Source workflow: {SOURCE_SKILL}
"""

from __future__ import annotations

import shutil
import signal
import sys
import time
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path

from superclaude.cli.pipeline.gates import gate_passed

from .config import load_{CLI_NAME}_config
from .diagnostics import DiagnosticCollector, FailureClassifier, ReportGenerator
from .logging_ import {PREFIX}Logger
from .models import (
    {PREFIX}Config,
    {PREFIX}MonitorState,
    {PREFIX}Outcome,
    {PREFIX}Result,
    {PREFIX}Status,
    {PREFIX}StepResult,
)
from .monitor import OutputMonitor, detect_error_max_turns
from .process import {PREFIX}Process, SignalHandler
from .tui import {PREFIX}TUI


def execute_{CLI_NAME}(config: {PREFIX}Config) -> {PREFIX}Result:
    """Execute the {PIPELINE_NAME} pipeline with supervised monitoring."""

    # Pre-flight
    if not shutil.which("claude"):
        print("Error: 'claude' binary not found in PATH", file=sys.stderr)
        sys.exit(1)

    # Setup infrastructure
    handler = SignalHandler()
    logger = {PREFIX}Logger(config)
    tui = {PREFIX}TUI(config)
    monitor = OutputMonitor()
    result = {PREFIX}Result(config=config)

    logger.write_header()
    tui.start()

    try:
        for step in _build_steps(config):
            if handler.shutdown_requested:
                result.outcome = {PREFIX}Outcome.INTERRUPTED
                break

            # Reset monitor for this step's output
            output_path = config.work_dir / step.output_file
            monitor.reset(output_path)
            monitor.start()

            # Launch subprocess
            process = {PREFIX}Process(config, step)
            logger.write_step_start(step)
            deadline = time.monotonic() + step.timeout_seconds
            process.start()

            # Supervision loop
            while process.is_running():
                if handler.shutdown_requested:
                    process.stop()
                    break

                if time.monotonic() > deadline:
                    process.stop()
                    break

                state = monitor.get_state()

                # Stall watchdog
                if (config.stall_timeout
                        and state.stall_seconds > config.stall_timeout
                        and state.events_received > 0):
                    if config.stall_action == "kill":
                        process.stop()
                        break

                try:
                    tui.update(step, state)
                except Exception:
                    pass

                time.sleep(0.5)

            # Classify result
            exit_code = process.wait()
            monitor.stop()
            status = _determine_status(exit_code, step, config)

            step_result = {PREFIX}StepResult(
                step=step,
                status=status,
                exit_code=exit_code,
                started_at=process.started_at,
                finished_at=time.time(),
                output_bytes=monitor.state.output_bytes,
            )
            result.step_results.append(step_result)
            logger.write_step_result(step_result)

            # Gate check for passing steps
            if status == {PREFIX}Status.PASS and step.gate:
                output_content = (config.work_dir / step.output_file).read_text()
                passed, reason = gate_passed(output_content, step.gate)
                if not passed:
                    step_result.status = {PREFIX}Status.HALT
                    step_result.gate_failure_reason = reason
                    status = {PREFIX}Status.HALT

            # Handle failures
            if status not in ({PREFIX}Status.PASS, {PREFIX}Status.PASS_NO_SIGNAL, {PREFIX}Status.PASS_NO_REPORT):
                collector = DiagnosticCollector()
                bundle = collector.collect(step_result, monitor.state, config)
                classifier = FailureClassifier()
                category = classifier.classify(bundle)
                reporter = ReportGenerator()
                reporter.write_report(bundle, category, config)

                result.outcome = {PREFIX}Outcome.HALTED
                result.halt_step = step.id
                break

        # Finalize
        if result.outcome == {PREFIX}Outcome.SUCCESS:
            all_passed = all(
                r.status in ({PREFIX}Status.PASS, {PREFIX}Status.PASS_NO_SIGNAL, {PREFIX}Status.PASS_NO_REPORT, {PREFIX}Status.SKIPPED)
                for r in result.step_results
            )
            if not all_passed:
                result.outcome = {PREFIX}Outcome.ERROR

        result.finished_at = time.time()
        logger.write_summary(result)

    finally:
        tui.stop()
        monitor.stop()
        handler.restore()

    return result


{BUILD_STEPS_FUNCTION}


def _determine_status(
    exit_code: int | None,
    step,
    config: {PREFIX}Config,
) -> {PREFIX}Status:
    """Classify step outcome from exit code and output artifacts."""
    if exit_code == 124:
        return {PREFIX}Status.TIMEOUT
    if exit_code and exit_code != 0:
        return {PREFIX}Status.ERROR

    result_path = config.work_dir / f"{step.id}-result.md"
    output_path = config.work_dir / step.output_file

    if result_path.exists():
        content = result_path.read_text()
        if "EXIT_RECOMMENDATION: HALT" in content:
            return {PREFIX}Status.HALT
        if "EXIT_RECOMMENDATION: CONTINUE" in content:
            return {PREFIX}Status.PASS
        if "status: PASS" in content.lower():
            return {PREFIX}Status.PASS
        if "status: FAIL" in content.lower():
            return {PREFIX}Status.HALT
        return {PREFIX}Status.PASS_NO_SIGNAL

    if output_path.exists():
        if detect_error_max_turns(output_path):
            return {PREFIX}Status.INCOMPLETE
        return {PREFIX}Status.PASS_NO_REPORT

    return {PREFIX}Status.ERROR
```
